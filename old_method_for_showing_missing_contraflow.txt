osmfilter interpreter.osm --keep="oneway=yes oneway:bicycle=yes" --drop-ways="oneway:bicycle=no" --drop-ways="service=turning_loop =driveway =parking_aisle" --drop-ways="junction=roundabout"  --drop-ways="dual_carriageway=yes" --drop-ways="dual_carriageway:bicycle=yes" --drop-ways="highway=primary =secondary" --drop-relations > oneway.osm

osmfilter oneway.osm --keep="highway=primary =secondary =tertiary =residential =unclassified =track =living_street =service" > relevant_oneway.osm
osmfilter relevant_oneway.osm --drop-ways="access=private =no" --drop-ways="bicycle=no =private =use_sidepath" > accessible_relevant_oneway.osm


osmfilter accessible_relevant_oneway.osm --keep="name" > named_relevant_oneway.osm

osmfilter named_relevant_oneway.osm --drop="name=Przegorzalska =Kapitana\ Mieczysława\ Medweckiego =Aleja\ Żubrowa =Aleja\ Kasy\ Oszczędności\ Miasta\ Krakowa =Ofiar\ Dąbia =Słotna =Biskupa\ Jana\ Prandoty =Józefa\ Marcika =Cystersów =Centralna =Mieczysława\ Medweckiego =Doktora\ Judyma =Rzemieślnicza =Franciszka\ Bujaka =Plac\ Braci\ Dudzińskich =Leonida\ Teligi =Mieczysławy\ Ćwiklińskiej =Bieżanowska =Prądnicka =Józefa\ Dietla =Wielicka =Na\ Kozłówce =Henryka\ Kamieńskiego" --keep-tags="all name highway=primary =secondary =tertiary =residential =unclassified =track =living_street =service" --drop-author > final_oneway.osm

osmfilter final_oneway.osm --keep="highway=primary =secondary =tertiary =residential =unclassified =track =living_street =service" > really_final_oneway.osm
